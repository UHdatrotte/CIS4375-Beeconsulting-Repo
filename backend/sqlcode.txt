# Source code for db schema
CREATE TABLE PROJECT(
PROJECT_ID INT AUTO_INCREMENT PRIMARY KEY,
CLIENT_FNAME varchar(100) NOT NULL,
CLIENT_LNAME varchar(100) NOT NULL,
COMPANY_NAME varchar(100) NOT NULL,
CLIENT_EMAIL varchar(150) NOT NULL,
CLIENT_PHONE char(12) NOT NULL,
COMPANY_STREET varchar(100) NOT NULL,
COMPANY_CITY varchar(50) NOT NULL,
COMPANY_STATE char(2) NOT NULL,
COMPANY_ZIP varchar(10) NOT NULL,
CONTRACT_TYPE varchar(10) NOT NULL,
START_DATE char(10) NOT NULL,
DURATION varchar(15) NOT NULL,
CONTRACT_RATE varchar(20) NOT NULL,
POSITION_REQ text NOT NULL,
JOB_STREET varchar(100) NOT NULL,
JOB_CITY varchar(50) NOT NULL,
JOB_STATE char(2) NOT NULL,
JOB_ZIP varchar(10) NOT NULL,
MARKETING text,
PROJECT_DESCRIPTION text,
SUBMITTED_AT datetime
);

CREATE TABLE TESTIMONIAL(
TESTIMONIAL_ID INT AUTO_INCREMENT PRIMARY KEY,
FNAME varchar(100) NOT NULL,
LNAME varchar(100) NOT NULL,
COMPANY_NAME varchar(100) NOT NULL,
EMAIL varchar(150) NOT NULL,
REVIEW text NOT NULL
);

CREATE TABLE TESTIMONIAL_AUDIT(
TESTIMONIAL_AUDIT_ID INT AUTO_INCREMENT PRIMARY KEY,
TESTIMONIAL_ID INT NOT NULL,
ADMIN_ID INT NOT NULL,
TIME_STAMP DATETIME DEFAULT CURRENT_TIMESTAMP,
ACTION_TYPE varchar(10),
FOREIGN KEY (TESTIMONIAL_ID) REFERENCES TESTIMONIAL(TESTIMONIAL_ID),
FOREIGN KEY (ADMIN_ID) REFERENCES ADMIN(ADMIN_ID)
);

CREATE TABLE PROJECT_AUDIT(
PROJECT_AUDIT_ID INT AUTO_INCREMENT PRIMARY KEY,
PROJECT_ID INT NOT NULL,
ADMIN_ID INT NOT NULL,
TIME_STAMP DATETIME DEFAULT CURRENT_TIMESTAMP,
ACTION_TYPE varchar(10),
FOREIGN KEY (PROJECT_ID) REFERENCES PROJECT(PROJECT_ID),
FOREIGN KEY (ADMIN_ID) REFERENCES ADMIN(ADMIN_ID)
);

CREATE TABLE ADMIN(
ADMIN_USER INT AUTO_INCREMENT PRIMARY KEY,
ADMIN_USER VARCHAR(100),
ADMIN_PASS_HASH VARCHAR(255)
)
